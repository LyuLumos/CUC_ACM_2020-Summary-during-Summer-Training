 [Codeforces Subsequences](https://vjudge.net/problem/CodeForces-1368B)

构造

对于“codeforces"字符串 添加任意一个位置为i（从0开始）的字符增添的子序列的个数为 (2^i)*(1^(9-i))

vector记录每个位置的字符出现次数

 

```
#include<bits/stdc++.h> 
using namespace std;
int main(){
	long long k,t=1;
	cin>>k;
	string s="codeforces";
	vector<int> cnt(10,1);
	for(int i=2;t<k;i++){
		for(int j=0;j<10;j++){
			if(t<k){
				t=t/cnt[j]*i;
				cnt[j]++;
			}
		}
	}
	for(int i=0;i<10;i++) cout<<string(cnt[i],s[i]);
	return 0;
}
```

## 

 [AND, OR and square sum](https://vjudge.net/problem/CodeForces-1368D)

位运算 贪心

 找规律可以发现 a&b 和 a|b 各位上一的数目和与 a,b 各位上1数目的和相同 要使平方和最大，则将二进制中1往高位放

```
#include<bits/stdc++.h> 
using namespace std;
typedef long long LL;
int cnt[30];
void solve(int num)
{
	for(int i=0;i<30;i++)
		cnt[i]+=((num>>i)&1);
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		int num;
		scanf("%d",&num);
		solve(num);
	}
	LL ans=0;
	for(int i=1;i<=n;i++)
	{
		LL num=0;
		for(int j=0;j<30;j++)
		{
			if(cnt[j])
			{
				cnt[j]--;
				num|=(1<<j);
			}
		}
		ans+=num*num;
	}
	printf("%lld\n",ans);
    return 0;
}
```
