```c++
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=1e5+5;
int main()
{
    int n,t;
    scanf("%d",&t);
    while(t--){
        ll mx=0,sum=0,x;
        scanf("%d",&n);
        for(int i=1;i<=n;i++){
            scanf("%lld",&x);
            sum+=x;
            mx=max(mx,(sum-1)/i+1);
        }
        printf("%lld\n",mx);
    }
	return 0;
}
```
